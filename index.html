    // Game State
    let gameState = {
    running: false,
    paused: false,
    score: 0,
    pilot: 'Guest',
    sfxVol: 0.7,
    musicVol: 0.5,

    let snake = [];
    let food = {};
    let direction = 'right';
    let nextDirection = 'right';

    if(gameType === 'snake') {
    document.getElementById('modalTitle').innerText = 'Move Snake v1.0';

    settingsPanel.style.display = 'block';
    gameContainer.style.display = 'none';

    document.getElementById('sfxVol').value = 70;
    document.getElementById('musicVol').value = 50;

    function startGame() {
    // Get Settings
    const nameInput = document.getElementById('pilotName').value.trim();
    gameState.pilot = nameInput || 'Anonymous';
    gameState.sfxVol = document.getElementById('sfxVol').value / 100;
    gameState.musicVol = document.getElementById('musicVol').value / 100;
    const mode = document.getElementById('displayMode').value;
    gameState.fullscreen = (mode === 'fullscreen');
    // UI Switch
    settingsPanel.style.display = 'none';
    gameContainer.style.display = 'block';
    pilotEl.innerText = `Pilot: ${gameState.pilot}`;
    // Init Audio
    initAudio();
    // Setup Canvas
    if (gameState.fullscreen) {
    canvas.width = window.innerWidth - 40;
    canvas.height = window.innerHeight - 100;
    if (document.documentElement.requestFullscreen) {
    document.documentElement.requestFullscreen().catch(e => console.log(e));
    }
    } else {
    canvas.width = 800;
    canvas.height = 600;
    }
    // Init Game Logic
    if (window.currentGame === 'strike') {
    initStrikeGame();
    } else {
    initSnakeGame();
    }
    gameState.running = true;
    gameState.paused = false;
    lastTime = performance.now();
    requestAnimationFrame(gameLoop);
    playStartSound();
    startBackgroundMusic();
    }
    function stopGame() {
    gameState.running = false;
    cancelAnimationFrame(animationFrameId);
    stopBackgroundMusic();
    if (document.fullscreenElement) {
    document.exitFullscreen().catch(e => console.log(e));
    }
    }
    function togglePause() {
    if (!gameState.running) return;
    gameState.paused = !gameState.paused;
    if (!gameState.paused) {
    lastTime = performance.now();
    requestAnimationFrame(gameLoop);
    } else {
    playPauseSound();
    }
    }
    // --- Snake Game Logic ---
    function initSnakeGame() {
    const gridSize = 20;
    const cols = Math.floor(canvas.width / gridSize);
    const rows = Math.floor(canvas.height / gridSize);
    snake = [
    {x: Math.floor(cols/2), y: Math.floor(rows/2)},
    {x: Math.floor(cols/2)-1, y: Math.floor(rows/2)},
    {x: Math.floor(cols/2)-2, y: Math.floor(rows/2)}
    ];
    direction = 'right';
    nextDirection = 'right';
    gameState.score = 0;
    scoreEl.innerText = `Score: 0`;
    spawnFood(gridSize, cols, rows);
    }
    function spawnFood(gridSize, cols, rows) {
    let valid = false;
    while(!valid) {
    food = {
    x: Math.floor(Math.random() * cols),
    y: Math.floor(Math.random() * rows)
    };
    valid = !snake.some(segment => segment.x === food.x && segment.y === food.y);
    }
    }
    function gameLoop(currentTime) {
    if (!gameState.running) return;
    if (gameState.paused) {
    animationFrameId = requestAnimationFrame(gameLoop);
    return;
    }
    const deltaTime = currentTime - lastTime;
    lastTime = currentTime;
    if (currentTime - lastMoveTime > speed) {
    updateSnake();
    lastMoveTime = currentTime;
    }
    drawSnake();
    animationFrameId = requestAnimationFrame(gameLoop);
    }
    function updateSnake() {
    direction = nextDirection;
    const head = { ...snake[0] };
    const gridSize = 20; // logical grid size for calculation, drawn scaled
    // Calculate grid dimensions based on current canvas
    const cols = Math.floor(canvas.width / gridSize);
    const rows = Math.floor(canvas.height / gridSize);
    if (direction === 'right') head.x++;
    if (direction === 'left') head.x--;
    if (direction === 'up') head.y--;
    if (direction === 'down') head.y++;
    // Wall Collision (Wrap around or Die? Let's Die for Noir style)
    if (head.x < 0 || head.x >= cols || head.y < 0 || head.y >= rows) {
    gameOver();
    return;
    }
    // Self Collision
    if (snake.some(seg => seg.x === head.x && seg.y === head.y)) {
    gameOver();
    return;
    }
    snake.unshift(head);
    // Eat Food
    if (head.x === food.x && head.y === food.y) {
    gameState.score += 10;
    scoreEl.innerText = `Score: ${gameState.score}`;
    playEatSound();
    spawnFood(gridSize, cols, rows);
    // Speed up slightly
    if(speed > 50) speed -= 2;
    } else {
    snake.pop();
    }
    }
    function drawSnake() {
    // Clear
    ctx.fillStyle = '#050505';
    ctx.fillRect(0, 0, canvas.width, canvas.height);
    // Draw Grid (Subtle)
    ctx.strokeStyle = '#1a1a1a';
    ctx.lineWidth = 1;
    const gridSize = 20;
    for(let x=0; x<canvas.width; x+=gridSize) {
    ctx.beginPath(); ctx.moveTo(x,0); ctx.lineTo(x,canvas.height); ctx.stroke();
    }
    for(let y=0; y<canvas.height; y+=gridSize) {
    ctx.beginPath(); ctx.moveTo(0,y); ctx.lineTo(canvas.width,y); ctx.stroke();
    }
    // Draw Food
    ctx.fillStyle = '#f4d03f';
    ctx.shadowBlur = 15;
    ctx.shadowColor = '#f4d03f';
    ctx.fillRect(food.x * gridSize + 2, food.y * gridSize + 2, gridSize - 4, gridSize - 4);
    ctx.shadowBlur = 0;
    // Draw Snake
    ctx.fillStyle = '#fff';
    snake.forEach((seg, index) => {
    if(index === 0) ctx.fillStyle = '#f4d03f'; // Head
    else ctx.fillStyle = '#ccc';
    ctx.fillRect(seg.x * gridSize + 1, seg.y * gridSize + 1, gridSize - 2, gridSize - 2);
    });
    }
    function gameOver() {
    gameState.running = false;
    playDieSound();
    stopBackgroundMusic();
    ctx.fillStyle = 'rgba(0,0,0,0.7)';
    ctx.fillRect(0, 0, canvas.width, canvas.height);
    ctx.fillStyle = '#f4d03f';
    ctx.font = '40px Courier New';
    ctx.textAlign = 'center';
    ctx.fillText('SYSTEM FAILURE', canvas.width/2, canvas.height/2 - 20);
    ctx.fillStyle = '#fff';
    ctx.font = '20px Courier New';
    ctx.fillText(`Final Score: ${gameState.score}`, canvas.width/2, canvas.height/2 + 20);
    ctx.fillText('Click Close to Restart', canvas.width/2, canvas.height/2 + 50);
    }
    // --- Input Handling ---
    document.addEventListener('keydown', (e) => {
    if (!gameState.running || gameState.paused) return;
    switch(e.key) {
    case 'ArrowUp': if(direction !== 'down') nextDirection = 'up'; break;
    case 'ArrowDown': if(direction !== 'up') nextDirection = 'down'; break;
    case 'ArrowLeft': if(direction !== 'right') nextDirection = 'left'; break;
    case 'ArrowRight': if(direction !== 'left') nextDirection = 'right'; break;
    case ' ': togglePause(); break;
    }
    });
    // Touch handling for mobile
    window.handleTouch = (dir) => {
    if (!gameState.running || gameState.paused) return;
    if(dir === 'up' && direction !== 'down') nextDirection = 'up';
    if(dir === 'down' && direction !== 'up') nextDirection = 'down';
    if(dir === 'left' && direction !== 'right') nextDirection = 'left';
    if(dir === 'right' && direction !== 'left') nextDirection = 'right';
    };
    window.handleTouchEnd = () => {};
    // --- Audio System (Synthesizer) ---
    function initAudio() {
    if(isAudioInit) return;
    const AudioContext = window.AudioContext || window.webkitAudioContext;
    audioCtx = new AudioContext();
    isAudioInit = true;
    }
    function playTone(freq, type, duration, vol) {
    if(!audioCtx) return;
    const osc = audioCtx.createOscillator();
    const gain = audioCtx.createGain();
    osc.type = type;
    osc.frequency.setValueAtTime(freq, audioCtx.currentTime);
    gain.gain.setValueAtTime(vol * gameState.sfxVol, audioCtx.currentTime);
    gain.gain.exponentialRampToValueAtTime(0.01, audioCtx.currentTime + duration);
    osc.connect(gain);
    gain.connect(audioCtx.destination);
    osc.start();
    osc.stop(audioCtx.currentTime + duration);
    }
    function playStartSound() {
    playTone(440, 'square', 0.1, 0.3);
    setTimeout(() => playTone(880, 'square', 0.2, 0.3), 100);
    }
    function playEatSound() {
    playTone(1200, 'sine', 0.1, 0.2);
    }
    function playDieSound() {
    playTone(150, 'sawtooth', 0.5, 0.5);
    setTimeout(() => playTone(100, 'sawtooth', 0.5, 0.5), 200);
    }
    function playPauseSound() {
    playTone(300, 'triangle', 0.1, 0.2);
    }
    let bgmInterval;
    function startBackgroundMusic() {
    if(!audioCtx) return;
    // Simple noir bassline
    const notes = [110, 110, 130, 110, 98, 98, 110, 130];
    let idx = 0;
    if(bgmInterval) clearInterval(bgmInterval);
    bgmInterval = setInterval(() => {
    if(gameState.paused) return;
    playTone(notes[idx], 'triangle', 0.4, gameState.musicVol * 0.5);
    idx = (idx + 1) % notes.length;
    }, 500);
    }
    function stopBackgroundMusic() {
    if(bgmInterval) clearInterval(bgmInterval);
    }

    let strikeState = {};
    function initStrikeGame() {
    const W = canvas.width;
    const H = canvas.height;
    strikeState = {
    player: { x: W/2, y: H*0.8, w: 28, h: 36, speed: 4, iframes: 0 },
    bullets: [],
    enemies: [],
    particles: [],
    stars: [],
    score: 0,
    lives: 3,
    wave: 1,
    frame: 0,
    fireTimer: 0,
    waveTimer: 0,
    waveActive: false,
    keys: {}
    };
    // Create stars
    for (let i = 0; i < 80; i++) {
    strikeState.stars.push({
    x: Math.random()*W,
    y: Math.random()*H,
    r: Math.random()*1.2+0.3,
    sp: Math.random()*0.3+0.1
    });

    spawnWave(1);

    document.addEventListener('keydown', handleStrikeKeydown);

    function spawnWave(w) {
    const count = 3 + w * 2;
    const cols = Math.min(count, 6);
    const rows = Math.ceil(count / cols);
    let spawned = 0;
    const W = canvas.width;
    for (let r = 0; r < rows && spawned < count; r++) {
    for (let c = 0; c < cols && spawned < count; c++) {
    const type = w >= 3 && Math.random() < 0.25 ? 'tank' : (w >= 2 && Math.random() < 0.3 ? 'fast' : 'basic');
    strikeState.enemies.push({
    x: (c + 0.5) * (W / cols),
    y: -60 - r * 70,
    w: type === 'tank' ? 34 : 26,
    h: type === 'tank' ? 28 : 22,
    hp: type === 'tank' ? 3 : 1,
    maxHp: type === 'tank' ? 3 : 1,
    speed: (type === 'fast' ? 1.8 : 1.1) + w * 0.15,
    type,
    angle: 0,
    shootTimer: Math.random() * 120,
    bullets: []
    });
    spawned++;
    }
    }
    }
    function updateStrikeGame() {
    const W = canvas.width;
    const H = canvas.height;
    const s = strikeState;
    s.frame++;
    ctx.clearRect(0, 0, W, H);
    // Background
    ctx.fillStyle = '#050508';
    ctx.fillRect(0, 0, W, H);
    // Stars
    for (const star of s.stars) {
    star.y += star.sp;
    if (star.y > H) { star.y = 0; star.x = Math.random()*W; }
    ctx.fillStyle = `rgba(245,197,24,${0.2 + star.r * 0.25})`;
    ctx.beginPath(); ctx.arc(star.x, star.y, star.r, 0, Math.PI*2); ctx.fill();
    }
    // Player movement
    const sp = s.player.speed;
    if (s.keys['ArrowLeft'] || s.keys['a'] || s.keys['A']) s.player.x = Math.max(s.player.w/2, s.player.x - sp);
    if (s.keys['ArrowRight'] || s.keys['d'] || s.keys['D']) s.player.x = Math.min(W - s.player.w/2, s.player.x + sp);
    if (s.keys['ArrowUp'] || s.keys['w'] || s.keys['W']) s.player.y = Math.max(60, s.player.y - sp);
    if (s.keys['ArrowDown'] || s.keys['s'] || s.keys['S']) s.player.y = Math.min(H - 30, s.player.y + sp);
    // Auto fire
    s.fireTimer++;
    if (s.fireTimer >= 10) {
    s.bullets.push({ x: s.player.x, y: s.player.y - 18, vy: -10 });
    s.fireTimer = 0;
    }
    if (s.player.iframes > 0) s.player.iframes--;
    // Update bullets
    for (let i = s.bullets.length - 1; i >= 0; i--) {
    const b = s.bullets[i];
    b.y += b.vy;
    if (b.y < -10) { s.bullets.splice(i, 1); continue; }
